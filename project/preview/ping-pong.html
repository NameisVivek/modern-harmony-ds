function getPP(id) { if (!ppState[id]) ppState[id] = { availSel: new Set(), selSel: new Set() }; return ppState[id]; } function selectItem(li, ppId) { const list = li.closest('.pp-items'); const side = list.id.includes('avail') ? 'availSel' : 'selSel'; const state = getPP(ppId); li.classList.toggle('selected-row'); const label = li.querySelector('.pp-item-label').textContent; if (li.classList.contains('selected-row')) state[side].add(label); else state[side].delete(label); updateButtons(ppId); } function updateButtons(ppId) { const state = getPP(ppId); const availList = document.getElementById(ppId + '-avail-list'); const selList = document.getElementById(ppId + '-sel-list'); const hasAvail = availList.children.length > 0; const hasSel = selList.children.length > 0; const availHighlighted = state.availSel.size > 0; const selHighlighted = state.selSel.size > 0; document.getElementById(ppId + '-add-all').disabled = !hasAvail; document.getElementById(ppId + '-add').disabled = !availHighlighted; document.getElementById(ppId + '-remove').disabled = !selHighlighted; document.getElementById(ppId + '-remove-all').disabled = !hasSel; document.getElementById(ppId + '-avail-count').textContent = availList.querySelectorAll('li:not([style*="display: none"])').length; document.getElementById(ppId + '-sel-count').textContent = selList.querySelectorAll('li:not([style*="display: none"])').length; } function moveSelected(ppId, dir) { const state = getPP(ppId); const fromId = dir === 'add' ? ppId + '-avail-list' : ppId + '-sel-list'; const toId = dir === 'add' ? ppId + '-sel-list' : ppId + '-avail-list'; const side = dir === 'add' ? 'availSel' : 'selSel'; const from = document.getElementById(fromId); const to = document.getElementById(toId); Array.from(from.querySelectorAll('.pp-item.selected-row')).forEach(li => { li.classList.remove('selected-row'); // Update check icon const check = li.querySelector('.pp-check .material-icons'); if (check) check.textContent = dir === 'add' ? 'check' : 'arrow_forward'; to.appendChild(li); }); state[side].clear(); restripeList(from); restripeList(to); updateButtons(ppId); } function moveAll(ppId, dir) { const fromId = dir === 'add' ? ppId + '-avail-list' : ppId + '-sel-list'; const toId = dir === 'add' ? ppId + '-sel-list' : ppId + '-avail-list'; const state = getPP(ppId); const from = document.getElementById(fromId); const to = document.getElementById(toId); Array.from(from.querySelectorAll('li')).forEach(li => { li.classList.remove('selected-row'); const check = li.querySelector('.pp-check .material-icons'); if (check) check.textContent = dir === 'add' ? 'check' : 'arrow_forward'; to.appendChild(li); }); state.availSel.clear(); state.selSel.clear(); restripeList(from); restripeList(to); updateButtons(ppId); } function restripeList(ul) { Array.from(ul.children).forEach((li, i) => { li.classList.toggle('alt', i % 2 === 1); }); } function filterList(ppId, side, query) { const listId = ppId + '-' + (side === 'available' ? 'avail' : 'sel') + '-list'; const q = query.toLowerCase(); document.getElementById(listId).querySelectorAll('li').forEach(li => { const label = li.querySelector('.pp-item-label').textContent.toLowerCase(); li.style.display = label.includes(q) ? '' : 'none'; }); updateButtons(ppId); } // Init button states updateButtons('pp1');
